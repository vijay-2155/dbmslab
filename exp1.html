Aim

To understand and implement basic SQL DDL commands for creating, modifying and deleting database tables, along with implementing constraints and practicing SELECT queries.

Theory

In Oracle SQL, DDL commands are used to define and modify the structure of database objects. The main commands covered in this experiment are CREATE TABLE, ALTER TABLE, and DROP TABLE along with various constraints.

Key Constraints:

  • PRIMARY KEY
  • FOREIGN KEY
  • NOT NULL
  • UNIQUE
  • CHECK

Syntax


CREATE TABLE table_name (
    column1 datatype constraint,
    column2 datatype constraint
);

ALTER TABLE table_name
ADD column_name datatype;

DROP TABLE table_name;

Sample Database Setup

-- Creating Tables


CREATE TABLE DEPARTMENT (
    DEPT_ID NUMBER PRIMARY KEY,
    DEPT_NAME VARCHAR2(20) NOT NULL,
    LOCATION VARCHAR2(30)
);

CREATE TABLE EMPLOYEE (
    EMP_ID NUMBER PRIMARY KEY,
    FIRST_NAME VARCHAR2(20) NOT NULL,
    LAST_NAME VARCHAR2(20),
    SALARY NUMBER CHECK (SALARY > 0),
    DEPT_ID NUMBER,
    FOREIGN KEY (DEPT_ID) REFERENCES DEPARTMENT(DEPT_ID)
);

-- Sample Queries


-- Basic SELECT with JOIN
SELECT E.FIRST_NAME, E.LAST_NAME, D.DEPT_NAME, E.SALARY
FROM EMPLOYEE E
JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID
ORDER BY E.SALARY DESC;

-- Group By Example
SELECT D.DEPT_NAME, COUNT(*) as EMP_COUNT, AVG(E.SALARY) as AVG_SALARY
FROM EMPLOYEE E
JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID
GROUP BY D.DEPT_NAME;

-- Basic SELECT Commands


-- Select all columns from DEPARTMENT table
SELECT * FROM DEPARTMENT;

-- Select specific columns from EMPLOYEE table
SELECT FIRST_NAME, LAST_NAME, SALARY FROM EMPLOYEE;

-- Using WHERE clause
SELECT * FROM EMPLOYEE WHERE SALARY > 55000;

-- Using ORDER BY
SELECT FIRST_NAME, SALARY FROM EMPLOYEE ORDER BY SALARY DESC;

-- Using DISTINCT
SELECT DISTINCT DEPT_ID FROM EMPLOYEE;

-- Using WHERE with multiple conditions
SELECT * FROM EMPLOYEE WHERE SALARY BETWEEN 45000 AND 60000;

-- Using LIKE operator
SELECT * FROM EMPLOYEE WHERE FIRST_NAME LIKE 'J%';

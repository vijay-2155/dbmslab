Example Queries

String Function Examples


-- String manipulation examples
SELECT 
    UPPER('hello world') as uppercase_text,
    LOWER('HELLO WORLD') as lowercase_text,
    INITCAP('hello world') as proper_case,
    LENGTH('hello world') as text_length,
    SUBSTR('hello world', 1, 5) as substring_text,
    LPAD('hello', 10, '*') as left_padded,
    RPAD('hello', 10, '*') as right_padded,
    LTRIM('   hello   ') as left_trimmed,
    RTRIM('   hello   ') as right_trimmed,
    INSTR('hello world', 'o') as position_of_o,
    'Hello' || ' ' || 'World' as concatenated
FROM dual;

Date Function Examples


-- Date manipulation examples
SELECT 
    SYSDATE as current_date,
    NEXT_DAY(SYSDATE, 'FRIDAY') as next_friday,
    ADD_MONTHS(SYSDATE, 3) as three_months_later,
    LAST_DAY(SYSDATE) as month_last_day,
    MONTHS_BETWEEN(
        TO_DATE('2025-12-31', 'YYYY-MM-DD'),
        TO_DATE('2025-01-01', 'YYYY-MM-DD')
    ) as months_diff,
    TRUNC(SYSDATE) as truncated_date,
    ROUND(SYSDATE) as rounded_date,
    LEAST(
        TO_DATE('2025-01-01', 'YYYY-MM-DD'),
        TO_DATE('2024-12-31', 'YYYY-MM-DD')
    ) as earlier_date,
    GREATEST(
        TO_DATE('2025-01-01', 'YYYY-MM-DD'),
        TO_DATE('2024-12-31', 'YYYY-MM-DD')
    ) as later_date
FROM dual;

Conversion Function Examples


-- Data type conversion examples
SELECT 
    -- Number to Character conversions
    TO_CHAR(1234.56, '99,999.99') as formatted_number,
    TO_CHAR(1234.56, '$9,999.99') as currency_format,
    
    -- Date to Character conversions
    TO_CHAR(SYSDATE, 'DD-MON-YYYY') as formatted_date,
    TO_CHAR(SYSDATE, 'Day, DD Month YYYY') as long_date,
    TO_CHAR(SYSDATE, 'HH24:MI:SS') as time_format,
    
    -- Character to Number conversion
    TO_NUMBER('1,234.56', '9,999.99') as parsed_number,
    
    -- Character to Date conversion
    TO_DATE('2025-01-01', 'YYYY-MM-DD') as parsed_date,
    TO_DATE('01-JAN-2025 14:30:00', 'DD-MON-YYYY HH24:MI:SS') as parsed_datetime
FROM dual;

Syntax Reference

String Function Syntax


-- Basic String Functions
UPPER(string)
LOWER(string)
INITCAP(string)
LENGTH(string)
SUBSTR(string, start_position [, length])
INSTR(string, substring [, start_position [, occurrence]])

-- Padding and Trimming
LPAD(string, padded_length [, pad_string])
RPAD(string, padded_length [, pad_string])
LTRIM(string [, trim_string])
RTRIM(string [, trim_string])
TRIM([trim_char FROM] string)

Date Function Syntax


-- Basic Date Functions
SYSDATE
NEXT_DAY(date, day_of_week)
ADD_MONTHS(date, number_of_months)
LAST_DAY(date)
MONTHS_BETWEEN(date1, date2)

-- Date Rounding
TRUNC(date [, format_model])
ROUND(date [, format_model])

-- Date Comparison
LEAST(date1, date2, ...)
GREATEST(date1, date2, ...)

Conversion Function Syntax


-- Number and Date to String
TO_CHAR(number [, format_model])
TO_CHAR(date [, format_model])

-- String to Number
TO_NUMBER(string [, format_model])

-- String to Date
TO_DATE(string [, format_model])

Common Format Models


-- Number Format Models
'9999.99'       -- Basic number with decimals
'$9,999.99'     -- Currency format
'09999'         -- Zero-padded number

-- Date Format Models
'YYYY-MM-DD'    -- Year-Month-Day
'DD-MON-YYYY'   -- Day-Month-Year
'HH24:MI:SS'    -- 24-hour time
'Day, DD Month YYYY' -- Full date format
